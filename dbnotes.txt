Table pokemon {
  id int [pk, increment]
  species varchar [not null]
  classification varchar [not null]
  type1 type [not null]
  type2 type
  height float [not null]
  weight float [not null]
  gender_ratio float [note: "% pokemon is male"]
  egg1 egg_group [not null]
  egg2 egg_group
  legendary bool
}

Table learns {
  poke_id int [ref: > pokemon.id]
  move_id int [ref: > moves.id]
  method varchar
  level int
}

Table moves {
  id int [pk, increment]
  name varchar [not null]
  type type [not null]
  category damage_category
  pp int
  power int
  accuracy float
  description varchar
  tm int [ref: - item.id]
}

Table obtained {
  poke_id int [ref: > pokemon.id]
  location int [ref: > locations.id]
  method obtain_method
}

Table evolves {
  prevo int [ref: > pokemon.id]
  evo int [ref: > pokemon.id]
  method varchar
  level int
  item int [ref: - item.id]
}

Table locations {
  id int [pk, increment]
  name varchar
}

Table entries {
  id int [pk, increment]
  poke_id int [ref: > pokemon.id]
  entry varchar
}

Table item {
  id int [pk, increment]
  purchase_price int
  sale_price int
  location int [ref: < locations.id]

}
